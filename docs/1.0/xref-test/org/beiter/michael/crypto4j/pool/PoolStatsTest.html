34  package org.beiter.michael.crypto4j.pool;
35  
36  import org.junit.Test;
37  
38  import static org.hamcrest.Matchers.equalTo;
39  import static org.hamcrest.Matchers.is;
40  import static org.junit.Assert.assertThat;
41  
42  public class PoolStatsTest {
43  
44      /**
45       * Test that the getters return the correct values
46       */
47      @Test
48      public void gettersTest() {
49  
50          PoolStats poolStats = getTestObject();
51  
52          // test all the getters
53          String error = "The returned value does not meet the expected value";
54          assertThat(error, poolStats.getNumActive(), is(equalTo(11)));
55          assertThat(error, poolStats.getNumIdle(), is(equalTo(12)));
56          assertThat(error, poolStats.getNumWaiters(), is(equalTo(13)));
57          assertThat(error, poolStats.getCountBorrowed(), is(equalTo(14L)));
58          assertThat(error, poolStats.getCountCreated(), is(equalTo(15L)));
59          assertThat(error, poolStats.getCountDestroyed(), is(equalTo(16L)));
60          assertThat(error, poolStats.getCountDestroyedByBorrowValidation(), is(equalTo(17L)));
61          assertThat(error, poolStats.getCountDestroyedByEvictor(), is(equalTo(18L)));
62          assertThat(error, poolStats.getCountReturned(), is(equalTo(19L)));
63          assertThat(error, poolStats.getMeanActiveTimeMillis(), is(equalTo(20L)));
64          assertThat(error, poolStats.getMeanBorrowWaitTimeMillis(), is(equalTo(21L)));
65          assertThat(error, poolStats.getMeanIdleTimeMillis(), is(equalTo(22L)));
66      }
67  
68      /**
69       * Test that the toString() method produces a correctly formatted String representation
70       */
71      @Test
72      public void stringTest() {
73  
74          PoolStats poolStats = getTestObject();
75  
76          StringBuilder expected = new StringBuilder();
77  
78          expected.append("Pool statistics:").append("\n")
79                  .append("\tnumActive: ").append(11).append("\n")
80                  .append("\tnumIdle: ").append(12).append("\n")
81                  .append("\tnumWaiters: ").append(13).append("\n")
82                  .append("\tcountBorrowed: ").append(14).append("\n")
83                  .append("\tcountCreated: ").append(15).append("\n")
84                  .append("\tcountDestroyed: ").append(16).append("\n")
85                  .append("\tcountDestroyedByBorrowValidation: ").append(17).append("\n")
86                  .append("\tcountDestroyedByEvictor: ").append(18).append("\n")
87                  .append("\tcountReturned: ").append(19).append("\n")
88                  .append("\tmeanActiveTimeMillis: ").append(20).append("\n")
89                  .append("\tmeanBorrowWaitTimeMillis: ").append(21).append("\n")
90                  .append("\tmeanIdleTimeMillis: ").append(22).append("\n");
91  
92          String error = "The returned value does not meet the expected value";
93          assertThat(error, poolStats.toString(), is(equalTo(expected.toString())));
94      }
95  
96      /**
97       * A private method that creates a test object
98       */
99      private static PoolStats getTestObject() {
100 
101         return new PoolStats(
102                 11, // numActive
103                 12, // numIdle
104                 13, // numWaiters
105                 14, // countBorrowed
106                 15, // countCreated
107                 16, // countDestroyed
108                 17, // countDestroyedByBorrowValidation
109                 18, // countDestroyedByEvictor
110                 19, // countReturned
111                 20, // meanActiveTimeMillis
112                 21, // meanBorrowWaitTimeMillis
113                 22  // meanIdleTimeMillis
114         );
115     }
116 }
