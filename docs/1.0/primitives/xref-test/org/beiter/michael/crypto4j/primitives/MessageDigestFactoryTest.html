188     /**
189      * Retrieve two singleton instances of {@code MessageDigest}, and assert that the two returned objects are identical
190      * (i.e. the factory returns a singleton).
191      * <p>
192      * Then, a regular (non-singleton) instance is retrieved, which are asserted to be different than the previously
193      * retrieved objects.
194      * <p>
195      * Finally, the factory is reset, and another instance is retrieved. If the factory resets properly, the third
196      * instance must be unequal to the first three instances.
197      */
198     @Test
199     public void factoryReturnsSingletonTest() {
200 
201         MessageDigestSpec spec = new MessageDigestSpec();
202 
203         // test that two singletons retrieved from the factory are identical
204         MessageDigest messageDigest1, messageDigest2;
205         try {
206             messageDigest1 = MessageDigestFactory.getSingleton(spec);
207             messageDigest2 = MessageDigestFactory.getSingleton(spec);
208         } catch (FactoryException | NoSuchAlgorithmException e) {
209             AssertionError ae = new AssertionError("Instantiation error");
210             ae.initCause(e);
211             throw ae;
212         }
213 
214         String error = "The factory does not return a singleton";
215         assertThat(error, messageDigest1, is(sameInstance(messageDigest2)));
216 
217         // then test that a regular (non-singleton) instance is different
218         MessageDigest messageDigest3;
219         try {
220             messageDigest3 = MessageDigestFactory.getInstance(spec);
221         } catch (FactoryException | NoSuchAlgorithmException e) {
222             AssertionError ae = new AssertionError("Instantiation error");
223             ae.initCause(e);
224             throw ae;
225         }
226         error = "The factory returns a singleton instead of a new object";
227         assertThat(error, messageDigest1, is(not(sameInstance(messageDigest3))));
228         assertThat(error, messageDigest2, is(not(sameInstance(messageDigest3))));
229 
230         // reset the factory
231         MessageDigestFactory.reset();
232 
233         // now test that the factory return a new object (i.e. a new singleton)
234         MessageDigest messageDigest4;
235         try {
236             messageDigest4 = MessageDigestFactory.getSingleton(spec);
237         } catch (FactoryException | NoSuchAlgorithmException e) {
238             AssertionError ae = new AssertionError("Instantiation error");
239             ae.initCause(e);
240             throw ae;
241         }
242 
243         error = "The factory does not return a singleton, or does not reset properly";
244         assertThat(error, messageDigest1, is(not(sameInstance(messageDigest4))));
245         assertThat(error, messageDigest2, is(not(sameInstance(messageDigest4))));
246         assertThat(error, messageDigest3, is(not(sameInstance(messageDigest4))));
247     }

279     /**
280      * Test that the pool factory method returns two different instances
281      * if called multiple times with the same pool properties
282      */
283     @Test
284     public void factoryReturnsMultiplePooledInstancesTest() {
285 
286         MessageDigestSpec mdSpec = new MessageDigestSpec();
287         PoolSpec poolSpec = new PoolSpec();
288         poolSpec.setMaxTotal(POOL_MAX_SIZE);
289         poolSpec.setMaxWaitMillis(0); // fail with an exception if no connections are available in the pool
290 
291         MessageDigest messageDigest1, messageDigest2;
292         try {
293             messageDigest1 = MessageDigestFactory.getPooledInstance(mdSpec, poolSpec);
294             messageDigest2 = MessageDigestFactory.getPooledInstance(mdSpec, poolSpec);
295         } catch (FactoryException | NoSuchAlgorithmException e) {
296             AssertionError ae = new AssertionError("Instantiation error");
297             ae.initCause(e);
298             throw ae;
299         }
300 
301         String error = "The first object returned by the pool is null";
302         assertThat(error, messageDigest1, is(notNullValue()));
303         error = "The second object returned by the pool is null";
304         assertThat(error, messageDigest2, is(notNullValue()));
305         error = "The first object returned by the pool is identical to the second object";
306         assertThat(error, messageDigest2, is(not(sameInstance(messageDigest1))));
307 
308         // the pool supports only 2 objects (see POOL_MAX_SIZE)
309         // borrowing a third instance will result in a FactoryException because the pool is exhausted
310         // we return the first instance, and then should be able to borrow a third one - which should
311         // be identical to the first one!
312         MessageDigestFactory.returnPooledInstance(mdSpec, messageDigest1);
313 
314         MessageDigest messageDigest3;
315         try {
316             messageDigest3 = MessageDigestFactory.getPooledInstance(mdSpec, poolSpec);
317         } catch (FactoryException | NoSuchAlgorithmException e) {
318             AssertionError ae = new AssertionError("Instantiation error");
319             ae.initCause(e);
320             throw ae;
321         }
322 
323         error = "The third object returned by the pool is null";
324         assertThat(error, messageDigest3, is(notNullValue()));
325         error = "The third object returned by the pool is NOT identical to the first object";
326         assertThat(error, messageDigest3, is(sameInstance(messageDigest1)));
327 
328         // if we reset the factory, we should be able to borrow two new objects, which should
329         // be different from the two objects we still reference from the old pool
330         MessageDigestFactory.reset();
331         MessageDigest messageDigest4, messageDigest5;
332         try {
333             messageDigest4 = MessageDigestFactory.getPooledInstance(mdSpec, poolSpec);
334             messageDigest5 = MessageDigestFactory.getPooledInstance(mdSpec, poolSpec);
335         } catch (FactoryException | NoSuchAlgorithmException e) {
336             AssertionError ae = new AssertionError("Instantiation error");
337             ae.initCause(e);
338             throw ae;
339         }
340 
341         error = "The fourth object returned by the pool is null";
342         assertThat(error, messageDigest4, is(notNullValue()));
343         error = "The fith object returned by the pool is null";
344         assertThat(error, messageDigest5, is(notNullValue()));
345         error = "The fourth object returned by the pool is identical to the fifth object";
346         assertThat(error, messageDigest4, is(not(sameInstance(messageDigest5))));
347         error = "The fourth object returned by the pool is identical to the second object";
348         assertThat(error, messageDigest4, is(not(sameInstance(messageDigest2))));
349         error = "The fourth object returned by the pool is identical to the third object";
350         assertThat(error, messageDigest4, is(not(sameInstance(messageDigest3))));
351         error = "The fifth object returned by the pool is identical to the second object";
352         assertThat(error, messageDigest5, is(not(sameInstance(messageDigest2))));
353         error = "The fifth object returned by the pool is identical to the third object";
354         assertThat(error, messageDigest5, is(not(sameInstance(messageDigest3))));
355 
356 
357         MessageDigestFactory.returnPooledInstance(mdSpec, messageDigest4);
358         MessageDigestFactory.returnPooledInstance(mdSpec, messageDigest5);
359     }

361     /**
362      * Test that the pool factory method does not return more more instances than are available in the pool
363      * <p>
364      * Note that this test is prone to resource leaks under certain conditions, which result in the borrowed
365      * connections not being properly returned to the pool. This is still okay for the unit tests, because
366      * the any pool resource leak does not longer matter after the tests are complete.
367      *
368      * @throws FactoryException When the instantiation does not work (expected)
369      */
370     @Test(expected = FactoryException.class)
371     public void factoryReturnsMultiplePooledInstancesFromExhaustedPoolTest()
372             throws FactoryException {
373 
374         MessageDigestSpec mdSpec = new MessageDigestSpec();
375         PoolSpec poolSpec = new PoolSpec();
376         poolSpec.setMaxTotal(POOL_MAX_SIZE);
377         poolSpec.setMaxWaitMillis(0); // fail with an exception if no connections are available in the pool
378 
379         MessageDigest messageDigest1, messageDigest2;
380         try {
381             messageDigest1 = MessageDigestFactory.getPooledInstance(mdSpec, poolSpec);
382             messageDigest2 = MessageDigestFactory.getPooledInstance(mdSpec, poolSpec);
383         } catch (FactoryException | NoSuchAlgorithmException e) {
384             AssertionError ae = new AssertionError("Instantiation error");
385             ae.initCause(e);
386             throw ae;
387         }
388 
389         String error = "The first object returned by the pool is null";
390         assertThat(error, messageDigest1, is(notNullValue()));
391         error = "The second object returned by the pool is null";
392         assertThat(error, messageDigest2, is(notNullValue()));
393         error = "The first object returned by the pool is identical to the second object";
394         assertThat(error, messageDigest2, is(not(sameInstance(messageDigest1))));
395 
396 
397         // the pool supports only 2 objects (see POOL_MAX_SIZE)
398         // borrowing a third instance will result in a FactoryException because the pool is exhausted
399         MessageDigest messageDigest3 = null;
400         try {
401             messageDigest1 = MessageDigestFactory.getPooledInstance(mdSpec, poolSpec);
402         } catch (NoSuchAlgorithmException e) {
403             AssertionError ae = new AssertionError("Instantiation error - unknown algorithm, this should never happen!");
404             ae.initCause(e);
405             throw ae;
406         } finally {
407 
408             if (messageDigest1 != null) {
409                 LOG.debug("returning resource 'messageDigest1'");
410                 MessageDigestFactory.returnPooledInstance(mdSpec, messageDigest1);
411                 LOG.debug("'messageDigest1' has been returned");
412             }
413 
414             if (messageDigest2 != null) {
415                 LOG.debug("returning resource 'messageDigest2'");
416                 MessageDigestFactory.returnPooledInstance(mdSpec, messageDigest2);
417                 LOG.debug("'messageDigest2' has been returned");
418             }
419 
420             if (messageDigest3 != null) {
421                 LOG.debug("returning resource 'messageDigest3'");
422                 MessageDigestFactory.returnPooledInstance(mdSpec, messageDigest3);
423                 LOG.debug("'messageDigest3' has been returned");
424             }
425         }
426     }

486     /**
487      * Test that the pool stats are correct
488      */
489     @Test
490     public void poolStatTest()
491             throws FactoryException, NoSuchAlgorithmException {
492 
493         MessageDigestSpec mdSpec = new MessageDigestSpec();
494         PoolSpec poolSpec = new PoolSpec();
495         poolSpec.setMaxTotal(POOL_MAX_SIZE);
496         poolSpec.setMaxWaitMillis(0); // fail with an exception if no connections are available in the pool
497 
498         // check pool utilization stats
499         MessageDigest messageDigest1, messageDigest2, messageDigest3;
500 
501         // first instance
502         messageDigest1 = MessageDigestFactory.getPooledInstance(mdSpec, poolSpec);
503         String error = "The pool utilization statistic is incorrect";
504         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getCountBorrowed(), is(equalTo(1L)));
505         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getCountReturned(), is(equalTo(0L)));
506         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getNumActive(), is(equalTo(1)));
507         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getNumIdle(), is(equalTo(0)));
508 
509         // second instance
510         messageDigest2 = MessageDigestFactory.getPooledInstance(mdSpec, poolSpec);
511         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getCountBorrowed(), is(equalTo(2L)));
512         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getCountReturned(), is(equalTo(0L)));
513         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getNumActive(), is(equalTo(2)));
514         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getNumIdle(), is(equalTo(0)));
515 
516         // return one instance
517         MessageDigestFactory.returnPooledInstance(mdSpec, messageDigest1);
518         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getCountBorrowed(), is(equalTo(2L)));
519         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getCountReturned(), is(equalTo(1L)));
520         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getNumActive(), is(equalTo(1)));
521         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getNumIdle(), is(equalTo(1)));
522 
523         // return the second instance
524         MessageDigestFactory.returnPooledInstance(mdSpec, messageDigest2);
525         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getCountBorrowed(), is(equalTo(2L)));
526         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getCountReturned(), is(equalTo(2L)));
527         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getNumActive(), is(equalTo(0)));
528         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getNumIdle(), is(equalTo(2)));
529 
530         // third instance
531         messageDigest3 = MessageDigestFactory.getPooledInstance(mdSpec, poolSpec);
532         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getCountBorrowed(), is(equalTo(3L)));
533         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getCountReturned(), is(equalTo(2L)));
534         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getNumActive(), is(equalTo(1)));
535         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getNumIdle(), is(equalTo(1)));
536 
537         // return the third instance
538         MessageDigestFactory.returnPooledInstance(mdSpec, messageDigest2);
539         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getCountBorrowed(), is(equalTo(3L)));
540         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getCountReturned(), is(equalTo(3L)));
541         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getNumActive(), is(equalTo(0)));
542         assertThat(error, MessageDigestFactory.getPoolStats(mdSpec).getNumIdle(), is(equalTo(2)));
543     }
