34  package org.beiter.michael.crypto4j.primitives.poolable;
35  
36  import org.apache.commons.lang3.Validate;
37  import org.apache.commons.pool2.BasePooledObjectFactory;
38  import org.apache.commons.pool2.PooledObject;
39  import org.apache.commons.pool2.impl.DefaultPooledObject;
40  import org.beiter.michael.crypto4j.primitives.spec.SecureRandomSpec;
41  import org.beiter.michael.crypto4j.primitives.SecureRandomFactory;
42  
43  import java.security.SecureRandom;
44  
45  /**
46   * This class provides a pool factory implementation for classes managed in this library to be used with the
47   * Apache Commons Pool, Version 2.
48   * <p>
49   * This class is for internal use of the library, and is not meant for public consumption. <b>Do not use this class
50   * outside of this library</b>, as it is not part of the library's public interface, and may be dropped / modified
51   * without warning in future releases.
52   */
53  public class SecureRandomPoolFactory
54          extends BasePooledObjectFactory<SecureRandom> {
55  
56      /**
57       * The object spec that this factory will use to generate objects.
58       */
59      private final SecureRandomSpec spec;
60  
61      /**
62       * Creates a factory and initializes it from an existing spec, making a defensive copy.
63       *
64       * @param spec The spec to use when creating new instances of the objects handled by this factory.
65       * @throws NullPointerException When {@code spec} is {@code null}
66       */
67      public SecureRandomPoolFactory(final SecureRandomSpec spec) {
68  
69          super();
70  
71          Validate.notNull(spec, "The validated object 'spec' is null");
72  
73          // create a defensive copy
74          this.spec = new SecureRandomSpec(spec);
75      }
76  
77      @Override
78      // Not our API - cannot fix...
79      @SuppressWarnings("PMD.SignatureDeclareThrowsException")
80      public final SecureRandom create()
81              throws Exception {
82  
83          return SecureRandomFactory.getInstance(spec);
84      }
85  
86      @Override
87      public final PooledObject<SecureRandom> wrap(final SecureRandom value) {
88  
89          return new DefaultPooledObject<>(value);
90      }
91  }
