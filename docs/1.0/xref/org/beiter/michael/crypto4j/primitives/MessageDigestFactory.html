90      /**
91       * Returns a new instance of {@link MessageDigest} with the provider and algorithm configured in the provided
92       * configuration object.
93       * <p>
94       * Note that the JCA interface for {@link MessageDigest} is <b>inherently not thread safe</b>, as each instance of
95       * {@link MessageDigest} maintains separate digests (i.e. the instances keep state based on previously processed
96       * data). Hence, when sharing the object returned by this method across several threads, utilize <b>appropriate
97       * locking mechanisms</b> in your code to prevent concurrent access.
98       *
99       * @param mdSpec The {@link MessageDigestSpec} to initialize the instance with
100      * @return An instance of {@link MessageDigest}
101      * @throws NullPointerException     When {@code mdSpec} is {@code null}
102      * @throws NoSuchAlgorithmException When the hash algorithm is not available
103      * @throws FactoryException         When the configured provider is not registered in the security provider list
104      */
105     public static MessageDigest getInstance(final MessageDigestSpec mdSpec)
106             throws NoSuchAlgorithmException, FactoryException {
107 
108         Validate.notNull(mdSpec, "The validated object 'mdSpec' is null");
109 
110         // compiler will optimize this:
111         final String providerName = mdSpec.getProviderName();
112         final String algorithmName = mdSpec.getAlgorithmName();
113 
114         return getInstance(providerName, algorithmName);
115     }

117     /**
118      * Returns a singleton instance of {@link MessageDigest} with the provider and algorithm configured in the provided
119      * configuration object. Note that different singletons are returned for permutations of the provided algorithm and
120      * provider settings.
121      * <p>
122      * Retrieving a singleton by this method will cause the factory to keep state, and store a reference to the
123      * singleton for later use. Use the singleton for low utilization use cases only, as there may be congestion
124      * otherwise. Getting a singleton object is slightly faster than getting a pooled object, and also reduces memory
125      * overhead, which makes using a singleton the go-forward approach unless the singleton cannot handle the
126      * computational load in multi-threaded applications.
127      * <p>
128      * <b>Additional notes</b>
129      * <ul>
130      * <li>The JCA {@link MessageDigest} is <b>inherently not thread safe</b>, as each instance maintains separate
131      * digests. Hence, when sharing the object returned by this method across several threads, utilize <b>appropriate
132      * locking mechanisms</b> in your code to prevent concurrent access.</li>
133      * <li>You may reset the factory state using the {@code reset()} method to retrieve a new / different singleton the
134      * next time this method is called.</li>
135      * <li>If you need tight control over the singleton, including its lifecycle and configuration, then you should
136      * create such objects with the {@code getInstance()} method and maintain their state as "singletons" in your
137      * application's business logic.</li>
138      * </ul>
139      *
140      * @param mdSpec The {@link MessageDigestSpec} to initialize the instance with
141      * @return An instance of {@link MessageDigest}
142      * @throws NullPointerException     When {@code mdSpec} is {@code null}
143      * @throws NoSuchAlgorithmException When the hash algorithm is not available
144      * @throws FactoryException         When the configured provider is not registered in the security provider list
145      */
146     // CHECKSTYLE:OFF
147     @SuppressWarnings("PMD.ConfusingTernary")
148     public static MessageDigest getSingleton(final MessageDigestSpec mdSpec)
149             throws NoSuchAlgorithmException, FactoryException {
150 
151         Validate.notNull(mdSpec, "The validated object 'mdSpec' is null");
152 
153         // this would be harder to read when following PMD's advice - ignoring the PMD and checkstyle warnings
154         //CHECKSTYLE:OFF checkstyle:avoidinlineconditionalscheck
155         final String providerName = (mdSpec.getProviderName() == null) ? "" : mdSpec.getProviderName();
156         //CHECKSTYLE:ON checkstyle:avoidinlineconditionalscheck
157         final String algorithmName = mdSpec.getAlgorithmName();
158 
159         final String key = getPoolKey(mdSpec);
160 
161         // create an instance if needed. The double-check idiom is not used for thread safety (the map already is
162         // thread safe), but to prevent unnecessary (expensive) object creation.
163         if (!SINGLETONS.containsKey(key)) {
164             synchronized (MessageDigestFactory.class) {
165                 if (!SINGLETONS.containsKey(key)) {
166 
167                     LOG.debug("Creating a new singleton for key '" + key + "'...");
168 
169                     SINGLETONS.putIfAbsent(key, getInstance(providerName, algorithmName));
170 
171                     LOG.debug("Successfully created a new singleton for key '" + key + "'");
172                 }
173             }
174         }
175 
176         return SINGLETONS.get(key);
177     }

179     /**
180      * Returns a pooled instance of {@link MessageDigest}. Initializing {@code MessageDigest} can be very time
181      * consuming. Using a {@code ThreadLocal} instance to reduce the impact of problem without introducing congestion
182      * on a shared object can be very problematic in an application server that reuses threads, and can lead to memory
183      * leaks and security issues (not to speak of the hassle to handle locking for the non-thread-safe JCA
184      * {@link MessageDigest} interface). To improve performance on object management, and avoid repeated JCA "create"
185      * calls, this method provides access to an object pool that manages shared object instances.
186      * <p>
187      * This method allows borrowing an object from a pool of {@link MessageDigest} with the provider and algorithm
188      * configured in the provided configuration object. In other words, when requesting an object of a specific
189      * configuration, this methods creates the necessary pooling infrastructure, properly configures the pool, and
190      * pre-populates it if needed. Note that different object configurations may use different object pools. However,
191      * <b>once a pool for a specific object configuration has been created, the pool and object configuration can not be
192      * changed without resetting this factory.</b> In other words, the pool spec for existing object specs is ignored in
193      * subsequent calls to this method, but a new object spec leads to creation of a new pool with that pool spec. See
194      * {@link MessageDigestFactory#reset()} on how to reset the factory.
195      * <p>
196      * <b>The object must be returned to the pool once it is not longer needed to prevent pool starvation!</b> See the
197      * {@link MessageDigestFactory#returnPooledInstance(MessageDigestSpec, MessageDigest)} method on how to return
198      * objects to the pool.
199      * <p>
200      * Retrieving an object by this method will cause the factory to keep state, and store a reference to the object
201      * pool for later use. Use the object pool to create a set of objects for sharing across multiple threads. This is
202      * less memory efficient than using a singleton, but allows an application to control object congestion. <b>Set
203      * the pool size to a value smaller than the number of concurrently active threads</b>, and use pooled objects over
204      * singletons in high utilization use cases. Getting a pooled object is slightly slower than getting a singleton.
205      * <p>
206      * The pool spec object controls how this factory manages the object pool. Chose the correct configuration and
207      * pool size (if needed) based on the expected utilization and congestion of the returned {@link MessageDigest}
208      * object pool in your application.
209      * <p>
210      * See {@link MessageDigestFactory#getSingleton(MessageDigestSpec)} for additional notes, in particular for the
211      * <b>thread safety requirements</b> that particularly apply when <b>sharing singletons across threads</b>.
212      *
213      * @param mdSpec   The spec to initialize the instances with
214      * @param poolSpec The spec to initialize the pool with
215      * @return A pooled instance of {@link MessageDigest}
216      * @throws NullPointerException     When {@code poolSpec} or {@code mdSpec} are {@code null}
217      * @throws NoSuchAlgorithmException When the hash algorithm is not available
218      * @throws FactoryException         When the configured provider is not registered in the security provider list,
219      *                                  or when the pool operation fails
220      * @throws IllegalStateException    When the pool is reset in parallel thread, while the current thread tries to
221      *                                  access it and borrow an object from the pool
222      */
223     // CHECKSTYLE:OFF
224     // this is flagged in checkstyle with a missing whitespace before '}', which is a bug in checkstyle
225     // We cannot remove the cause of the NPE without introducing complex synchronized locking. That would be a poor
226     // decision from a performance perspective, because the client still needs to handle the FactoryException anyway.
227     @SuppressWarnings({"PMD.AvoidCatchingNPE", "PMD.AvoidCatchingGenericException"})
228     // CHECKSTYLE:ON
229     public static MessageDigest getPooledInstance(final MessageDigestSpec mdSpec, final PoolSpec poolSpec)
230             throws NoSuchAlgorithmException, FactoryException {
231 
232         Validate.notNull(poolSpec, "The validated object 'poolSpec' is null");
233         Validate.notNull(mdSpec, "The validated object 'mdSpec' is null");
234 
235         // create an inbound defensive copy of the MessageDigestSpec
236         final MessageDigestSpec myMdSpec = new MessageDigestSpec(mdSpec);
237 
238         final String key = getPoolKey(myMdSpec);
239 
240         // create an instance if needed. The double-check idiom is not used for thread safety (the map already is
241         // thread safe), but to prevent unnecessary (expensive) object creation.
242         if (!OBJECT_POOL.containsKey(key)) {
243             synchronized (MessageDigestFactory.class) {
244                 if (!OBJECT_POOL.containsKey(key)) {
245 
246                     LOG.debug("Creating a new object pool for key '" + key + "'...");
247 
248                     // create a defensive copy of the configuration object for the pool
249                     final GenericObjectPoolConfig config = getPoolConfig(poolSpec);
250 
251                     // create a new pool for this key
252                     final GenericObjectPool<MessageDigest> pool
253                             = new GenericObjectPool<MessageDigest>(new MessageDigestPoolFactory(myMdSpec), config);
254 
255                     // store the pool in the map
256                     OBJECT_POOL.putIfAbsent(key, pool);
257 
258                     LOG.debug("Successfully created a new object pool for key '" + key + "'");
259                 }
260             }
261         }
262 
263         // get the correct pool, borrow an object, and return it
264         try {
265             // Note: the get() operation can lead to an NPE if the pool is reset at the same time when this method
266             // is executed. Any exception that would happen during the pool operation is swallowed by the pool.
267             return OBJECT_POOL.get(key).borrowObject();
268 
269         } catch (NullPointerException e) {
270             final String error = "Error when returning the object to the pool."
271                     + " Has the factory been reset in a parallel thread?";
272             LOG.warn(error);
273             throw new IllegalStateException(error, e);
274         } catch (Exception e) {
275             LOG.debug("Failing object pool ID: " + key);
276             final String error = "No object could be retrieved from the pool";
277             LOG.warn(error, e);
278             throw new FactoryException(error, e);
279         }
280     }

282     /**
283      * Return a pooled instance back to the pool.
284      * <p>
285      * Note that the pooled objects are standard JCA classes, and hence do not implement the {@link java.io.Closeable}
286      * interface. Use this method to return an object back to the pool for reuse.
287      * <p>
288      * As this factory may potentially manage several pools of objects with equal type that have been created with
289      * varying specifications, provide the object spec used to create the pooled object, thus allowing this method to
290      * identify the correct pool to return the object to.
291      *
292      * @param mdSpec       The object spec used to create the pooled object
293      * @param messageDigest The object to be returned to the pool
294      * @throws NullPointerException  When {@code mdSpec} or {@code messageDigest} are {@code null}
295      * @throws IllegalStateException If no pool exists for the provided object spec, or
296      *                               if an object is returned to to the wrong pool (i.e. the wrong {@code }mdSpec})
297      */
298     // CHECKSTYLE:OFF
299     // this is flagged in checkstyle with a missing whitespace before '}', which is a bug in checkstyle
300     // We cannot remove the cause of the NPE without introducing complex synchronized locking. That would be a poor
301     // decision from a performance perspective, because the client still needs to handle the FactoryException anyway.
302     @SuppressWarnings({"PMD.AvoidCatchingNPE", "PMD.AvoidCatchingGenericException"})
303     // CHECKSTYLE:ON
304     public static void returnPooledInstance(final MessageDigestSpec mdSpec, final MessageDigest messageDigest)
305             throws IllegalStateException {
306 
307         Validate.notNull(mdSpec, "The validated object 'mdSpec' is null");
308         Validate.notNull(messageDigest, "The validated object 'messageDigest' is null");
309 
310         // create an inbound defensive copy of the MessageDigestSpec
311         final MessageDigestSpec myMdSpec = new MessageDigestSpec(mdSpec);
312 
313         final String key = getPoolKey(myMdSpec);
314 
315         // return the object to the correct pool
316         if (OBJECT_POOL.containsKey(key)) {
317             try {
318                 // Note: the get() operation can lead to an NPE if the pool is reset at the same time when this method
319                 // is executed. Any exception that would happen during the pool operation is swallowed by the pool.
320                 OBJECT_POOL.get(key).returnObject(messageDigest);
321 
322             } catch (NullPointerException e) {
323                 final String error = "Error when returning the object to the pool."
324                         + " Has the factory been reset in a parallel thread?";
325                 LOG.warn(error);
326             }
327         } else {
328             LOG.debug("Failing object pool ID: " + key);
329             final String error = "The object pool does not exist";
330             LOG.warn(error);
331             throw new IllegalStateException(error);
332         }
333     }

337      * <p>

341      *
342      * @param mdSpec The object spec used to create the pooled object
343      * @return a {@code PoolStats} object with the usage statistics of the pool identified by the provided spec
344      * @throws NullPointerException  When {@code mdSpec} is {@code null}
345      * @throws IllegalStateException If no pool exists for the provided object spec

347     // CHECKSTYLE:OFF
348     // this is flagged in checkstyle with a missing whitespace before '}', which is a bug in checkstyle
349     // We cannot remove the cause of the NPE without introducing complex synchronized locking. That would be a poor
350     // decision from a performance perspective, because the client still needs to handle the FactoryException anyway.
351     @SuppressWarnings({"PMD.AvoidCatchingNPE", "PMD.AvoidCatchingGenericException"})
352     // CHECKSTYLE:ON
353     public static PoolStats getPoolStats(final MessageDigestSpec mdSpec) {
354 
355         Validate.notNull(mdSpec, "The validated object 'mdSpec' is null");
356 
357         // create an inbound defensive copy of the MessageDigestSpec
358         final MessageDigestSpec myMdSpec = new MessageDigestSpec(mdSpec);
359 
360         final String key = getPoolKey(myMdSpec);
361 
362         if (OBJECT_POOL.containsKey(key)) {
363             try {
364                 // Note: the get() operation can lead to an NPE if the pool is reset at the same time when this method

393             throw new IllegalStateException(error);
394         }

397     /**
398      * Resets the internal state of the factory. This includes resetting the pool, which causes the following methods
399      * to return new {@link MessageDigest} instances the next time they are called:
400      * <p>
401      * <ul>
402      * <li>{@link MessageDigestFactory#getSingleton(MessageDigestSpec)}</li>
403      * <li>{@link MessageDigestFactory#getPooledInstance(MessageDigestSpec, PoolSpec)}</li>
404      * </ul>
405      */
406     public static void reset() {
407 
408         // clear the singletons
409         if (SINGLETONS.size() > 0) {
410 
411             SINGLETONS.clear();
412 
413             LOG.debug("Singletons cleared");
414         }
415 
416         // clear the pools with all objects. The double-check idiom is not used for thread safety (the map already is
417         // thread safe), but to prevent unnecessary (expensive) object operations.
418         if (OBJECT_POOL.size() > 0) {
419             synchronized (MessageDigestFactory.class) {
420                 if (OBJECT_POOL.size() > 0) {
421 
422                     // Note: the final cleanup will be complete once all borrowed objects have been returned to the
423                     //       pools, but this class will no longer return any new instances from the pools.
424 
425                     // first close all pools...
426                     for (final Map.Entry<String, GenericObjectPool<MessageDigest>> entry : OBJECT_POOL.entrySet()) {
427                         entry.getValue().close();
428                         LOG.debug("Object pool for key '" + entry.getKey() + "' closed");
429                     }
430 
431                     // ...then clear the map, hence de-referencing all the pools
432                     OBJECT_POOL.clear();
433                     LOG.debug("Object pool cleared");
434                 }
435             }
436         }
437 
438     }

440     /**
441      * Returns an instance of MessageDigest that implements the specified hash algorithm.
442      * <p>
443      * This method returns a MessageDigest object encapsulating the MessageDigestSpi implementation from the specified
444      * provider. The specified provider must be registered in the security provider list.
445      * <p>
446      * If the specified provider is <code>null</code> or empty, this method traverses the list of registered security
447      * Providers, starting with the most preferred Provider. A new MessageDigest object encapsulating the
448      * MessageDigestSpi implementation from the first Provider that supports the configured hash algorithm is returned.
449      *
450      * @param providerName  The name of the provider
451      * @param algorithmName The name of the hash algorithm, as defined in the JCA Standard Algorithm Name Documentation
452      * @return Instance of MessageDigest
453      * @throws IllegalArgumentException When the provided value of {@code algorithmName} is null or empty
454      * @throws NoSuchAlgorithmException When the hash algorithm is not available*
455      * @throws FactoryException         When the specified provider is not registered in the security provider list
456      */
457     private static MessageDigest getInstance(final String providerName, final String algorithmName)
458             throws NoSuchAlgorithmException, FactoryException {
459 
460         // no formal validation - private method
461 
462         // we still check that we got an algorithm name that at least looks valid:
463         if (algorithmName == null || algorithmName.isEmpty()) {
464             final String error = "The message digest algorithm name cannot be null or empty";
465             LOG.warn(error);
466             throw new IllegalArgumentException(error);
467         }
468 
469         if (providerName == null || providerName.isEmpty()) {
470 
471             final String error = "The message digest provider name is not set, using the JCA default provider";
472             LOG.info(error);
473 
474             try {
475                 return MessageDigest.getInstance(algorithmName);
476             } catch (NoSuchAlgorithmException e) {
477                 final String error2 = "The provided hash algorithm '" + algorithmName + "' is not supported";
478                 LOG.warn(error2, e);
479                 throw new NoSuchAlgorithmException(error, e);
480             }
481         } else {
482 
483             try {
484                 return MessageDigest.getInstance(algorithmName, providerName);
485             } catch (NoSuchAlgorithmException e) {
486                 final String error = "The provided hash algorithm '" + algorithmName + "' is not supported";
487                 LOG.warn(error, e);
488                 throw new NoSuchAlgorithmException(error, e);
489             } catch (NoSuchProviderException e) {
490                 final String error = "The provided MessageDigest provider '" + providerName
491                         + "' is not registered in the security provider list";
492                 LOG.warn(error, e);
493                 throw new FactoryException(error, e);
494             }
495         }
496     }

498     /**
499      * Compute a key to uniquely reference objects of a specific spec (i.e. a spec that results in a different object
500      * behavior will produce a different key).
501      *
502      * @param mdSpec The object spec
503      * @return a key that uniquely identifies the object spec
504      */
505     @SuppressWarnings("PMD.ConfusingTernary")
506     private static String getPoolKey(final MessageDigestSpec mdSpec) {
507 
508         // this would be harder to read when following PMD's advice - ignoring the PMD and checkstyle warnings
509         //CHECKSTYLE:OFF checkstyle:avoidinlineconditionalscheck
510         final String providerName = (mdSpec.getProviderName() == null) ? "" : mdSpec.getProviderName();
511         //CHECKSTYLE:ON checkstyle:avoidinlineconditionalscheck
512         final String algorithmName = mdSpec.getAlgorithmName();
513 
514         // key for a MessageDigest are the provider and the algorithm
515         // (concatenation is good enough here)
516         final String key = providerName + algorithmName; // compiler will optimize this
517 
518         return key;
519     }

521     /**
522      * Returns a pool configuration object based on the provided pool spec
523      *
524      * @param poolSpec The pool spec with the pool's settings
525      * @return the pool configuration object
526      */
527     private static GenericObjectPoolConfig getPoolConfig(final PoolSpec poolSpec) {
528 
529         // no validation - private method
530 
531         final GenericObjectPoolConfig config = new GenericObjectPoolConfig();
532 
533         config.setMaxTotal(poolSpec.getMaxTotal());
534         config.setMaxIdle(poolSpec.getMaxIdle());
535         config.setMinIdle(poolSpec.getMinIdle());
536         config.setMaxWaitMillis(poolSpec.getMaxWaitMillis());
537         config.setTestOnCreate(poolSpec.isTestOnCreate());
538         config.setTestOnBorrow(poolSpec.isTestOnBorrow());
539         config.setTestOnReturn(poolSpec.isTestOnReturn());
540         config.setTestWhileIdle(poolSpec.isTestWhileIdle());
541         config.setTimeBetweenEvictionRunsMillis(poolSpec.getTimeBetweenEvictionRunsMillis());
542         config.setNumTestsPerEvictionRun(poolSpec.getNumTestsPerEvictionRun());
543         config.setMinEvictableIdleTimeMillis(poolSpec.getMinEvictableIdleTimeMillis());
544         config.setSoftMinEvictableIdleTimeMillis(poolSpec.getSoftMinEvictableIdleTimeMillis());
545         config.setEvictionPolicyClassName(poolSpec.getEvictionPolicyClassName());
546         config.setLifo(poolSpec.isLifo());
547         config.setFairness(poolSpec.isFairness());
548         config.setBlockWhenExhausted(poolSpec.isBlockWhenExhausted());
549         config.setJmxEnabled(poolSpec.isJmxEnabled());
550         config.setJmxNamePrefix(poolSpec.getJmxNamePrefix());
551         config.setJmxNameBase(poolSpec.getJmxNameBase());
552 
553         return config;
554     }
